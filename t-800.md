4030 LET DINERO=DINERO+INGRESO

4090 IF INGRESO>0 THEN PRINT "INGRESO PASIVO: +";INGRESO

7040 IF F<>1 OR DINERO<1000 THEN GOTO 100
7050 LET DINERO=DINERO-1000: LET DEUDA=DEUDA-1000
7070 IF DEUDA<=0 THEN GOTO 8000
7080 GOTO 4000

8000 CLS: PRINT "¡ERES LIBRE DE DEUDAS!": PRINT "EN EL DIA ";DIA: PAUSE 0: RUN

9000 REM === SUBRUTINA PERRO ===
9010 PRINT "": IF DINERO<50 THEN PRINT "PERRO: MAL...": RETURN
9020 IF DINERO>=50 AND DINERO<150 THEN PRINT "PERRO: SOBREVIVES."
9030 IF DINERO>=150 THEN PRINT "PERRO: VAS ENTENDIENDO EL JUEGO."
9040 RETURN
